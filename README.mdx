export const link =
info => {
    const [_, folder, name, desc] = /^(@?[\w.-]+\/)?([\w.-]+)(?: *: *(.+))?$/.exec(info)
    const host = `https://github.com`
    return ``
        + (folder
            ? `- [${name}](${host}/${
                folder.startsWith("@")
                    ? folder.slice(1)
                    : "gnlow-"+folder
            }${name})`
            : `- [${name}](${host}/gnlow/${name})`)
        + (desc ? `: ${desc}` : ``)
}

export const links =
([str]) =>
    str.trim().split("\n").map(x => link(x.trim())).join("\n")
